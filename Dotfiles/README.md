xinitrc --> ~/.xinitrc  
Xresources --> ~/.Xresources  
i3config --> ~/.config/i3/config  
i3statusconfig --> ~/config/i3status/config  
bashrc --> ~/.bashrc 
Xkbmap --> ~/.Xkbmap
